\dwfig{Figures/deepwiki/arquitectura_e2e.png}
  {Arquitectura extremo a extremo del sistema. La actividad fMRI de BOLD5000
   (ROIs visuales, \texttt{src/config.py::BOLD5000\_ROI\_SET}) alimenta el
   adapter fMRI$\rightarrow$CLIP (\texttt{phase2/train\_adapter.py}), cuyo
   embedding ViT-L/14 condiciona el generador SD~2.1 unCLIP
   (\texttt{src/sd\_decoder.py}) para producir la imagen reconstruida
   (\texttt{phase2/visual\_evaluator.py}).}
  {fig:arq_e2e}

\dwfig{Figures/deepwiki/flujo_modulos_phase2.png}
  {Flujo de datos y dependencias entre los módulos de \texttt{src/phase2/}:
   \texttt{bold5000\_loader} $\rightarrow$ \texttt{extract\_vit\_features}
   $\rightarrow$ \texttt{train\_adapter} $\rightarrow$ \texttt{visual\_evaluator}
   $\rightarrow$ \texttt{compare\_subjects} $\rightarrow$
   \texttt{build\_appendix\_montages}. Cada nodo corresponde a un script real
   del repositorio.}
  {fig:flujo_modulos}

\dwfig{Figures/deepwiki/mapa_modulos_repo.png}
  {Mapa de módulos del repositorio: \texttt{src/} (núcleo del pipeline),
   \texttt{tests/} (suite mock sin GPU), \texttt{exe/} (lanzadores) y
   \texttt{AlvaroTaipe\_Plantilla/} (fuentes LaTeX). La portabilidad
   dev/inferencia se resuelve por variables \texttt{ACECOM\_*}
   (\texttt{src/config.py}, \texttt{src/locate\_recons.py}).}
  {fig:mapa_repo}
